			k_x \\

	\end{aligned} \\ \\

		\begin{bmatrix}
			M_x \\

			M_{xy}
		\end{bmatrix}

			B_{11} & B_{12} & B_{16} \\
			B_{12} & B_{22} & B_{26} \\
